import pandas as pd

def union_two_df(df_users_activity: pd.DataFrame, df_user_data: pd.DataFrame):
    df_union = pd.merge(df_users_activity, df_user_data,
                        how='left', left_on=['user_id', 'login_date'],
                        right_on=['user_id', 'install_date'])
    print("first 10 rows in df: \n")
    print(df_union.head(10))
    print("column names in df: \n")
    print(df_union.columns)
    return df_union

def clean_df(df_union: pd.DataFrame):
    df_union['ua_channel'] = df_union.groupby('user_id')['ua_channel'].ffill().bfill()
    df_union['cpi'] = df_union.groupby('user_id')['cpi'].ffill().bfill()
    df_union['platform'] = df_union.groupby('user_id')['platform'].ffill().bfill()
    df_union['install_date'] = df_union.groupby('user_id')['install_date'].ffill().bfill()
    df_union['install_date'] = pd.to_datetime(df_union['install_date'])
    df_union['login_date'] = pd.to_datetime(df_union['login_date'])
    df_union['cohort_day'] = df_union['login_date'] - df_union['install_date']
    df_union['cohort_day'] = df_union['cohort_day'].dt.days
    df_union['cohort_day'] = df_union['cohort_day'].astype(int)
    df_union_cleaned = df_union[df_union['login_date'] >= df_union['install_date']]

    return df_union_cleaned

def add_cumulative_metrics(df_union: pd.DataFrame):
    df_union_uu = df_union.groupby(by=['install_date', 'login_date', 'platform', 'ua_channel', 'cohort_day'],
                                               as_index=False)['user_id'].nunique()
    df_union_uu.rename(columns={'user_id': 'user_count'}, inplace=True)


    df_union = df_union.groupby(['install_date', 'login_date', 'platform', 'ua_channel', 'cohort_day'], as_index=False).sum()
    df_union = pd.merge(df_union, df_union_uu, how='inner')

    df_union['ad_revenue_cum'] = df_union.groupby(by=['install_date', 'platform', 'ua_channel'])['ad_revenue'].cumsum()
    df_union['iap_revenue_cum'] = df_union.groupby(by=['install_date', 'platform', 'ua_channel'])['iap_revenue'].cumsum()
    df_union['transactions_cum'] = df_union.groupby(by=['install_date', 'platform', 'ua_channel'])['transactions'].cumsum()
    df_union['total_time_sec_cum'] = df_union.groupby(by=['install_date', 'platform', 'ua_channel'])['total_time_sec'].cumsum()
    df_union['missions_cum'] = df_union.groupby(by=['install_date', 'platform', 'ua_channel'])['missions'].cumsum()
    df_union['sessions_cum'] = df_union.groupby(by=['install_date', 'platform', 'ua_channel'])['sessions'].cumsum()

    df_union.to_csv("df_union.csv", index=False)
    return df_union

if __name__ == "__main__":
    df_user_data = pd.read_csv('df_user_data.csv')
    df_users_activity = pd.read_csv('df_users_activity.csv')
    df_union = union_two_df(df_users_activity, df_user_data)
    df_clean = add_cumulative_metrics(clean_df(df_union))

import pymc3 as pm
import numpy as np
import matplotlib.pyplot as plt

# generate sample data
days = np.arange(1, 31)
retention_rates = [0.8, 0.75, 0.7, 0.65, 0.6, 0.55, 0.5, 0.45, 0.4, 0.35,
                   0.3, 0.25, 0.2, 0.15, 0.1, 0.05, 0.05, 0.05, 0.05, 0.05,
                   0.05, 0.05, 0.05, 0.05, 0.05, 0.05, 0.05, 0.05, 0.05, 0.05]

n_users = 1000

# define the model
with pm.Model() as model:
    alpha = pm.Uniform("alpha", 0, 1)
    beta = pm.Uniform("beta", 0, 1)
    sigma = pm.Uniform("sigma", 0, 1)

    mu = alpha * np.power(beta, days)

    likelihood = pm.Normal("likelihood", mu=mu, sd=sigma, observed=retention_rates)

    # sample from the posterior
    trace = pm.sample(1000, chains=2)

# plot the results
pm.plot_posterior(trace, varnames=["alpha", "beta", "sigma"])
plt.show()
