import pandas as pd


def add_retention(df: pd.DataFrame):
    """
    Calculates the retention rates for different time intervals.

    :param df: Input data
    :return: Data with retention rates
    """
    intervals = [1, 3, 7, 14, 30, 60]
    for interval in intervals:
        dau_col = f"d{interval}_dau"
        ret_col = f"d{interval}_ret"
        df[ret_col] = df[dau_col] / df["installs"]

    return df

def add_arpu(df: pd.DataFrame):
    """
    Calculates the revenue per user different time intervals.

    :param df: Input data
    :return: Data with ARPU
    """
    intervals = [1, 3, 7, 14, 30, 60]
    for interval in intervals:
        revenue_col_iap = f"d{interval}_iap_revenue"
        revenue_col_ads = f"d{interval}_ads_revenue"
        roi_col = f"d{interval}_arpu"
        df[roi_col] = (df[revenue_col_iap] + df[revenue_col_ads]) / df["installs"]
    return df

def add_adsarpu(df: pd.DataFrame):
    """
    Calculates the ads revenue per user different time intervals.

    :param df: Input data
    :return: Data with AdsARPU

    """
    intervals = [1, 3, 7, 14, 30, 60]
    for interval in intervals:
        revenue_col_ads = f"d{interval}_ads_revenue"
        roi_col = f"d{interval}_adsarpu"
        df[roi_col] = (df[revenue_col_ads]) / df["installs"]
    return df

def add_iaparpu(df: pd.DataFrame):
    """
    Calculates the iap revenue per user for different time intervals.

    :param df: Input data
    :return: Data with IAPARPU

    """
    intervals = [1, 3, 7, 14, 30, 60]
    for interval in intervals:
        revenue_col_iap = f"d{interval}_iap_revenue"
        roi_col = f"d{interval}_iaparpu"
        df[roi_col] = (df[revenue_col_iap]) / df["installs"]
    return df


def add_revenue(df: pd.DataFrame):
    """
    Calculates the cumulative revenue for different time intervals.

    :param df: Input data
    :return: Data with Revenue
    """
    intervals = [1, 3, 7, 14, 30, 60]
    for interval in intervals:
        revenue_col_iap = f"d{interval}_iap_revenue"
        revenue_col_ads = f"d{interval}_ads_revenue"
        roi_col = f"d{interval}_revenue"
        df[roi_col] = (df[revenue_col_iap] + df[revenue_col_ads])
    return df

def add_arppu(df: pd.DataFrame):
    """
    Calculates the average revenue per paying users for different time intervals.

    :param df: Input data
    :return: Data with ARPPU
    """
    intervals = [1, 3, 7, 14, 30, 60]
    for interval in intervals:
        revenue_col_iap = f"d{interval}_iap_revenue"
        fts_col = f"d{interval}_first_time_spenders"
        ratio_col = f"d{interval}_arppu"
        df[ratio_col] = df[revenue_col_iap] / df[fts_col]

    return df

def add_fts(df: pd.DataFrame):
    """
    Calculates the ratio of first-time spenders to installs for different time intervals.

    :param df: Input data
    :return: Data with FTS ratio
    """
    intervals = [1, 3, 7, 14, 30, 60]
    for interval in intervals:
        revenue_col_iap = f"d{interval}_iap_revenue"
        fts_col = f"d{interval}_first_time_spenders"
        ratio_col = f"d{interval}_fts_ratio"
        df[ratio_col] = df[fts_col] / df["installs"]

    return df


def add_metrics(df: pd.DataFrame):
    """
    Adds retention, ROI, arpu, adsarpu, iaparpu and FTS ratio metrics to the dataframe.

    :param df: Input data
    :return: Data with retention, ROI, arpu, adsarpu, iaparpu and FTS ratio metrics
    """

    add_retention(df)
    add_arpu(df)
    add_arppu(df)
    add_adsarpu(df)
    add_iaparpu(df)
    add_fts(df)

    return df
