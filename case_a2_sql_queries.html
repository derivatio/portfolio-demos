#PostgreSQL
#Query 1:
#This query generates the table that I used Analyse of Allocated ROAS section.

WITH TABLE_ALL AS (
    SELECT 
        "Month", 
        "Spend" AS "Total Spend",
        "Installs" AS "Total Installs",
        "Blended Revenue D30" AS "Total Revenue D30"
    FROM public.anonymized_table 
    WHERE "Source" IN ('ALL') 
),

TABLE_SOURCE AS (
    SELECT 
        "Month", 
        "Source", 
        "Installs",
        "Spend"
    FROM public.anonymized_table 
    WHERE "Source" NOT IN ('ALL') 
),

TABLE_MERGE AS (
    SELECT 
        a.*, 
        s."Source",
        s."Installs",
        s."Spend"
    FROM TABLE_ALL a
    LEFT JOIN TABLE_SOURCE s 
    ON a."Month" = s."Month"
),

TABLE_FINAL AS (
  SELECT 
    "Month",
    "Source",
    sum("Spend") as "Spend",
    sum("Installs") as "Installs",
    sum("Spend") / sum(NULLIF("Installs",0)) AS "CPI",
    sum(NULLIF("Installs",0)) / sum(NULLIF("Total Installs",0)) * sum("Total Revenue D30") AS "Est. Revenue",
    (sum(NULLIF("Installs",0)) / sum(NULLIF("Total Installs",0)) * sum("Total Revenue D30")) / sum(NULLIF("Spend",0)) AS "Est. ROI"
  FROM TABLE_MERGE
  GROUP BY "Month","Source"
)

SELECT * FROM TABLE_FINAL

#Query 2:
#This query used for paid / organic installs graph
SELECT 
    "Month" AS "Month", 
    SUM(CASE 
        WHEN "Source" = 'Organic' THEN "Installs" 
        ELSE NULL 
    END) AS "Organic Installs",
    SUM(CASE 
        WHEN "Source" NOT IN ('Organic', 'ALL') THEN "Installs" 
        ELSE NULL 
    END) AS "Paid Installs"
FROM public.anonymized_table
GROUP BY "Month"
ORDER BY "Month" ASC

#Query 3:
#This query used for Analyse of 4 Months section
SELECT 
"Month" AS "Month", 
"Source" AS "Source", 
sum("Installs") AS "Total Installs", 
sum("Spend") AS "Total Spend",
sum("Spend") / sum("Installs") AS "CPI",
sum("Blended Revenue D30") / sum("Installs") AS "ARPU D30",
sum("Blended Revenue D30") / sum("Spend") AS "Blended ROI D30"
FROM public.anonymized_table 
GROUP BY "Month", "Source" ORDER BY "Month" ASC
