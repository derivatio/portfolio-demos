retention.sql

WITH TABLE_raw AS (
    SELECT
        MIN(CASE WHEN new_player = 'True' THEN date END) OVER (PARTITION BY user_id) AS install_date,
        DATE(date) AS date,
        game_name,
        country_code,
        os,
        COUNT(DISTINCT user_id) AS user_count,
        DATEDIFF('day', MIN(DATE((CASE new_player WHEN True THEN date END))) OVER (PARTITION BY user_id), DATE(date) ) AS cohort_day
        
    FROM
        data
    WHERE
        user_id IS NOT NULL
    GROUP BY
        install_date,
        date,
        game_name,
        country_code,
        os,
        cohort_day
),
TABLE_usercount AS (
    SELECT
        install_date,
        game_name,
        country_code,
        os,
        MAX(CASE cohort_day WHEN 0 THEN user_count END) AS user_d0,
        MAX(CASE cohort_day WHEN 1 THEN user_count END) AS user_d1,
        MAX(CASE cohort_day WHEN 3 THEN user_count END) AS user_d3,
        MAX(CASE cohort_day WHEN 7 THEN user_count END) AS user_d7
    FROM
        TABLE_raw
    WHERE
        cohort_day >= 0
    GROUP BY
        install_date,
        game_name,
        country_code,
        os
)
SELECT
    install_date,
    game_name,
    country_code,
    os,
    SUM(user_d1) / SUM(user_d0) AS retention_d1,
    SUM(user_d3) / SUM(user_d0) AS retention_d3,
    SUM(user_d7) / SUM(user_d0) AS retention_d7
FROM
    TABLE_usercount
GROUP BY
    install_date,
    game_name,
    country_code,
    os

import warnings
warnings.filterwarnings("ignore")

import pandas as pd
# Set the display format for floating-point numbers
# This line specifies that floating-point numbers should be displayed with two decimal places
pd.options.display.float_format = '{:,.2f}'.format

# Set the display option to show all columns
# This line sets the display option to show all columns in the DataFrame
pd.set_option('display.max_columns', None)

# Read the CSV files into dataframes
data_game_events = pd.read_csv("Game_started event.csv")
data_ad_revenue = pd.read_csv("Ad Revenue.csv")

# Function to create the install date based on new_player column
def create_installdate(data: pd.DataFrame):
    new_data = data.copy()
    new_data["install_date"] = new_data[new_data["new_player"] == True]["date"]
    new_data = new_data[new_data["install_date"].isna() == False]

    new_data = new_data[["user_id", "install_date"]]
    new_data = new_data.groupby(["user_id"], as_index=False).min()

    new_data.drop_duplicates(inplace=True)
    result = pd.merge(data, new_data, how="left")
    return result

# Function to create the cohort day based on install_date and date
def create_cohortday(data: pd.DataFrame):
    data["user_id"] = data["user_id"].astype(str)

    data["date"] = pd.to_datetime(data["date"])
    data["install_date"] = pd.to_datetime(data["install_date"])
    data["cohort_day"] = data["date"] - data["install_date"]
    data["cohort_day"] = data["cohort_day"].apply(lambda x: str(x).split(" ")[0])
    data = data[data["cohort_day"] != "NaT"]
    data["cohort_day"] = data["cohort_day"].astype(int)
    data = data[data["cohort_day"] >= 0]
    data.sort_values(by=["user_id"], inplace=True)

    return data

# Function to add retention metrics to the dataframe
def add_retention(data: pd.DataFrame):
    data = create_installdate(data)
    data = create_cohortday(data)

    data["user_id"] = data["user_id"].astype(str)

    day0 = pd.DataFrame(data[data["cohort_day"] == 0].groupby(by=["install_date", "date", "game_name", "os", "country_code"],
                                                            as_index=False). \
                       agg({"user_id": "nunique"}).rename(columns={"user_id": "user_d0"}))

    day1 = pd.DataFrame(data[data["cohort_day"] == 1].groupby(by=["install_date", "date", "game_name", "os", "country_code"],
                                                            as_index=False). \
                       agg({"user_id": "nunique"}).rename(columns={"user_id": "user_d1"}))

    day3 = pd.DataFrame(
        data[data["cohort_day"] == 3].groupby(by=["install_date", "date", "game_name", "os", "country_code"],
                                          as_index=False). \
        agg({"user_id": "nunique"}).rename(columns={"user_id": "user_d3"}))

    day7 = pd.DataFrame(data[data["cohort_day"] == 7].groupby(by=["install_date", "date", "game_name", "os", "country_code"],
                                                            as_index=False). \
                       agg({"user_id": "nunique"}).rename(columns={"user_id": "user_d7"}))

    day0.drop(columns=["date"], inplace=True)
    day1.drop(columns=["date"], inplace=True)
    day3.drop(columns=["date"], inplace=True)
    day7.drop(columns=["date"], inplace=True)

    df_last_1 = pd.merge(day0, day1, how='left', on=["install_date", "game_name", "os", "country_code"])
    df_last_2 = pd.merge(df_last_1, day3, how='left', on=["install_date", "game_name", "os", "country_code"])
    df_last_3 = pd.merge(df_last_2, day7, how='left', on=["install_date", "game_name", "os", "country_code"])

    df_last_3 = df_last_3.groupby(by=["install_date", "game_name", "os", "country_code"],
                                                            as_index=False).sum()

    df_last_3["retention_d1"] = df_last_3["user_d1"] / df_last_3["user_d0"]
    df_last_3["retention_d3"] = df_last_3["user_d3"] / df_last_3["user_d0"]
    df_last_3["retention_d7"] = df_last_3["user_d7"] / df_last_3["user_d0"]

    df_last_3.drop(columns=["user_d0", "user_d1", "user_d3", "user_d7"], inplace=True)
    df_last_3.drop_duplicates(inplace=True)
    df_last_3.rename(columns={"install_date": "date"},inplace=True)
    return df_last_3

# Function to filter and aggregate new player data
def add_newplayers(data: pd.DataFrame):
    data = data[data["new_player"] == True]
    data.drop_duplicates(inplace=True)
    new_user = pd.DataFrame(
        data.groupby(by=["date", "game_name", "os", "country_code"],
                                              as_index=False). \
        agg({"user_id": "nunique"}).rename(columns={"user_id": "user_new"}))
    new_user["date"] = pd.to_datetime(new_user["date"])

    return new_user

# Function to calculate daily active users (DAU)
def add_dau(data: pd.DataFrame):
    data.drop_duplicates(inplace=True)
    dau = pd.DataFrame(
        data.groupby(by=["date", "game_name", "os", "country_code"],
                                              as_index=False). \
        agg({"user_id": "nunique"}).rename(columns={"user_id": "dau"}))
    dau["date"] = pd.to_datetime(dau["date"])

    return dau

# Function to prepare the game event CSV data
def prepare_game_event_csv(data: pd.DataFrame):
    columns = ["date", "os", "country_code", "game_name", "new_player", "user_id"]
    data = data[columns]
    data["user_id"] = data["user_id"].astype(str)
    data = data[data["user_id"].isna() == False]
    data.drop_duplicates(inplace=True)
    data_retention = add_retention(data)
    data_newplayers = add_newplayers(data)
    data_dau = add_dau(data)
    merge1 = pd.merge(data_dau, data_newplayers, how='inner')
    result = pd.merge(merge1, data_retention, how='inner')

    return result

# Function to calculate the fill rate
def add_fillrate(data: pd.DataFrame):
    data["fill_rate"] = data["impressions"] / data["ad_requests"]
    return data


# Function to calculate the ARPDAU (Average Revenue Per Daily Active User)
def add_arpdau(data: pd.DataFrame):
    data["arpdau"] = data["revenue"] / data["dau"]
    return data

# Function to prepare the ad revenue CSV data
def prepare_adrevenue_csv(data: pd.DataFrame):
    data["date"] = pd.to_datetime(data["date"])
    data = add_fillrate(data)
    data["revenue"].fillna(0.0, inplace=True)
    data["ad_requests"].fillna(0, inplace=True)
    data["impressions"].fillna(0, inplace=True)
    return data

# Function to merge the game events and ad revenue dataframes
def merge_2df(data_game_event: pd.DataFrame, data_adrevenue: pd.DataFrame):
    data_game_event.rename(columns={"country_code": "country"}, inplace=True)
    result = pd.merge(data_game_event, data_adrevenue, how="left")
    result = add_arpdau(result)
    return result

# Main execution
if __name__ == '__main__':
    # Prepare the game event CSV data
    df_game_events = prepare_game_event_csv(data_game_events)
    # Prepare the ad revenue CSV data
    df_adrevenue = prepare_adrevenue_csv(data_ad_revenue)
    # Merge the two dataframes
    df_result = merge_2df(df_game_events, df_adrevenue)
    # Save the result to a CSV file
    df_result.to_csv("result.csv", index=False)
